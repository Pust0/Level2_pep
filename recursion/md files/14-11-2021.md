```java
public static int getNum(String s, HashMap<Character, Integer> charIntMap){
        String num="";
        for(int i=0; i<s.length(); ++i)
            num+=charIntMap.get(s.charAt(i));
        return Integer.parseInt(num);
    }

  public static void solution(String unique, int idx, HashMap<Character, Integer> charIntMap, boolean[] usedNumbers, String s1, String s2, String s3) {
    if(idx==unique.length()){
        int num1=getNum(s1, charIntMap);
        int num2=getNum(s2, charIntMap);
        int num3=getNum(s3, charIntMap);

        if(num1+num2==num3){
            for(int i=0; i<26; ++i){
                char ch=(char)('a'+i);
                if(charIntMap.containsKey(ch)){
                    System.out.print(ch+"-"+charIntMap.get(ch)+" ");
                }
            }
            System.out.println();
        }
        return;
    }

    char ch=unique.charAt(idx);
    for(int num=0; num<=9; ++num){
        if(!usedNumbers[num]){
            usedNumbers[num]=true;
            charIntMap.put(ch, num);
            solution(unique, idx+1, charIntMap, usedNumbers, s1, s2, s3);
            usedNumbers[num]=false;
            charIntMap.put(ch, -1);
        }
    }
}
```

## 2. Marks of PCM

```cpp
//code
#include<iostream>
#include<vector>
#include<bits/stdc++.h>
using namespace std;
class marks{
    public:
        int phy=0;
        int chem=0;
        int math=0;
        
        marks() {
            
        }
        
        marks(int phy,int chem,int math) {
            this->phy=phy;
            this->chem=chem;
            this->math=math;
        }
    };
    
class compare{
    public:
    bool operator()(marks& a, marks& b){
        if(a.phy!=b.phy){
                return a.phy<b.phy;
            }else if(a.chem!=b.chem){
                return a.chem>b.chem;
            }else{
                return a.math<b.math;
            }
    }
};



void customSort(vector<int> &phy, vector<int> &chem, vector<int> &math){
        int N = phy.size();
        vector<marks> arr;
        
        for(int i=0; i < N;i++) {
            marks m={phy[i],chem[i],math[i]};
            arr.push_back(m);
        }
        
        sort(arr.begin(),arr.end(),compare());
        
        for(int i=0; i < N;i++) {
            phy[i] = arr[i].phy;
            chem[i] = arr[i].chem;
            math[i] = arr[i].math;
        }
}

int main(){
    int n;
    cin>>n;
    vector<int> phy(n,0), chem(n,0), math(n,0);
    
    for(int i=0; i<n; ++i){
        cin>>phy[i];
    }
    for(int i=0; i<n; ++i){
        cin>>chem[i];
    }
    for(int i=0; i<n; ++i){
        cin>>math[i];
    }
    
    customSort(phy, chem, math);
    
    for(int i=0; i<n; ++i){
```

## 3. Find K closest elements
```cpp
#include<bits/stdc++.h>
#include<iostream>

using namespace std;

class Pair{
    public:
    int val=0;
    int gap=0;
    
    Pair(){
        
    }
    
    Pair(int val, int gap){
        this->val=val;
        this->gap=gap;
    }
};

class compare{
    public:
    bool operator()(const Pair &a, const Pair &b){
        if(a.gap==b.gap){
            return a.val < b.val;
        }
        return a.gap < b.gap;
    }
};

vector<int> findClosest(vector<int> &arr, int k, int x){
    vector<int> list;
    priority_queue<Pair,vector<Pair>,compare> pq;
    
    for(int i=0; i<arr.size(); ++i){
        if(pq.size()<k){
            Pair m={arr[i], abs(arr[i]-x)};
            pq.push(m);
        }else{
            if(pq.top().gap>abs(arr[i]-x)){
                pq.pop();
                Pair m={arr[i], abs(arr[i]-x)};
                pq.push(m);
            }
        }
    }
    
    while(pq.size()>0){
        int data=pq.top().val;
        pq.pop();
        list.push_back(data);
    }
    
    sort(list.begin(), list.end());
    
    return list;
}

int main(){
    int n;
    cin>>n;
    vector<int> arr;
    
    for(int i=0; i<n; ++i){
        int a;
        cin>>a;
        arr.push_back(a);
    }
    
    int k, x;
    cin>>k>>x;
    
    vector<int> ans=findClosest(arr, k, x);
    
    for(int ele: ans){
        cout<<ele<<" ";
    }
    
    return 0;
}
```

## 4.Median of two sorted arrays
```cpp
#include<bits/stdc++.h>
using namespace std;

float find(vector<float> &a1,vector<float> &a2) {
    //write your code here
    int n=a1.size()+a2.size();
    vector<float> merged(n);

    int i, j, k;
    i=j=k=0;

    while(i<a1.size() && j<a2.size()){
        if(a1[i]<a2[j]){
            merged[k]=a1[i];
            i++;
            k++;
        }else{
            merged[k]=a2[j];
            j++;
            k++;
        }
    }

    while(i<a1.size()){
        merged[k]=a1[i];
        i++;
        k++;
    }

    while(j<a2.size()){
        merged[k]=a2[j];
        j++;
        k++;
    }

    int mid=merged.size()/2;
    float median=0;
    if(merged.size()%2==0){
        median = (float)(merged[mid]+merged[mid-1])/2.0;
    }else{
        median = merged[mid];
    }

    return median;
}
```

## 5. Count the triplets
```cpp
#include<bits/stdc++.h>
using namespace std;

  int countTriplets(vector<int> &arr) {
    //write your code here
    sort(arr.begin(), arr.end());

    int count = 0;

    for (int i = arr.size() - 1; i >= 1; i--) {
      int l = 0;
      int r = i - 1;

      while (l < r) {
        if (arr[l] + arr[r] == arr[i]) {
          count++;
          l++;
          r--;
        }
        else if (arr[l] + arr[r] < arr[i]) {
          l++;
        }
        else {
          r--;
        }
      }
    }

    return count;
  }

  int main() {
    //input work
    int n;
    cin>>n;
    vector<int> arr(n);

    for (int i = 0; i < n; i++) {
        int data;
        cin>>data;
      arr[i] = data;
    }

    cout<<countTriplets(arr);
  }
```

## 6. Counting elements in two arrays
```cpp
#include<bits/stdc++.h>
using namespace std;

int helper(int val, vector<int> arr) {
    //count smaller elements than val in arr
    int count = 0;

    int lo = 0;
    int hi = arr.size() - 1;

    while (lo <= hi) {
      int mid = (lo + hi) / 2;

      if (arr[mid] <= val) {
        count += (mid - lo + 1);
        lo = mid + 1;
      } else {
        hi = mid - 1;
      }
    }

    return count;
  }

vector<int> find(vector<int> &arr1, vector<int> &arr2) {
    //write your code here
    vector<int> ans(arr1.size());
    sort(arr2.begin(), arr2.end());

    for (int i = 0; i < arr1.size(); i++) {
      int cs = helper(arr1[i], arr2);
      ans[i] = cs;
    }

    return ans;
  }

int main() {
    //input work
    int n1, n2;
    
    cin>>n1;
    vector<int> a(n1);
    for (int i = 0; i < n1; i++) {
      cin>>a[i];
    }

    cin>>n2;
    vector<int> b(n2);
    for (int i = 0; i < n2; i++) {
      cin>>b[i];
    }

    vector<int> ans = find(a, b);

    for (int val : ans) {
      cout<<val<<" ";
    }
}
```
